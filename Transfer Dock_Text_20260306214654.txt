        // Main calendar render
        function renderCalendar() {
            const grid = document.getElementById("calendar-grid");
            grid.innerHTML = "";
            
            const firstDay = new Date(currentYear, currentMonth, 1).getDay();
            const daysInMonth = new Date(currentYear, currentMonth + 1, 0).getDate();
            
            // Weekday headers already static, but we fill dates
            // Empty slots
            for (let i = 0; i < firstDay; i++) {
                const empty = document.createElement("div");
                empty.className = "calendar-day bg-transparent";
                grid.appendChild(empty);
            }
            
            // Actual days
            const today = new Date();
            const isCurrentMonth = (today.getFullYear() === currentYear && today.getMonth() === currentMonth);
            
            for (let day = 1; day <= daysInMonth; day++) {
                const dayEl = document.createElement("div");
                dayEl.className = `calendar-day day-cell flex flex-col`;
                
                // Day number
                const num = document.createElement("div");
                num.className = "calendar-day-number text-xl";
                num.textContent = day;
                dayEl.appendChild(num);
                
                // Check if has event
                const key = `${currentYear}-${String(currentMonth).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
                if (calendarEvents[key]) {
                    const eventContainer = document.createElement("div");
                    eventContainer.className = "flex-1 flex items-end";
                    const dot = document.createElement("div");
                    dot.className = "event-dot";
                    dot.style.backgroundColor = calendarEvents[key].color || "#f43f5e";
                    eventContainer.appendChild(dot);
                    dayEl.appendChild(eventContainer);
                }
                
                // Click handler
                dayEl.onclick = () => openEventModal(day);
                
                // Highlight today
                if (isCurrentMonth && today.getDate() === day) {
                    dayEl.style.border = "2px solid #eab308";
                    dayEl.style.borderRadius = "8px";
                }
                
                grid.appendChild(dayEl);
            }
            
            // Update header texts
            document.getElementById("preview-month").textContent = monthNames[currentMonth];
            document.getElementById("preview-year").textContent = currentYear;
            
            // Update header icon area
            const headerArea = document.getElementById("calendar-header");
            headerArea.innerHTML = `
                <div class="text-xs bg-zinc-100 text-zinc-500 font-medium px-5 py-3 rounded-3xl">2025</div>

                    <i class="fa-solid fa-chevron-left"></i>
                </div>
            `;
        }
